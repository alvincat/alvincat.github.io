# tar常用的命令参数
# -j 使用bzip处理文件
# -J 使用xz处理文件
# -z 使用gzip处理文件
# -c 压缩文件
# -x 解压文件
# -C 指定操作目录
# -v 显示正在处理的文件

# 示例1. 打包为bz文件
tar -jcvf test.tar.bz ./test/

# 示例2. 解压gz文件
tar -zxvf test.gz
